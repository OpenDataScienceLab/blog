#############################################################
STEP BY STEP PUBLISHING INTO OPENDATASCIENCE-BLOG
##############################################################

1 - #INSTALL PYTHON-PIP
$ sudo apt install python3.6

$ sudo apt install python3-pip

2 - #CREATE AND ACTIVATE CONDA_ENV PY3.6
$ conda create --name OpenDScience-Blog python=3.6

$ conda activate OpenDScience-Blog

3 - #CLONAR REPOSITORIO
$ git clone https://github.com/esloch/nk-blog.git

4 - #ABRIR O DIRETORIO
$ cd nk-blog

5 - #LOGAR NO BRANCH SOURCE MASTER
$ git checkout origin/master

6 - #INSTALLING EXTRAS, SETUPTOOLS, PIP, WEEL...

#Note:
Updating pip, setuptools, and wheel is recommended.
$ pip install --upgrade pip setuptools wheel

$ pip install --upgrade "Nikola[extras]"

7 - #INSTALAR JUPYTER
$ pip install jupyterlab notebook

8 - #COMPILAR O BLOG LOCAL
$ nikola build

9 - #CRIAR NOVO POST NORMAL
$ nikola new_post

10 - #CRIAR NOVO POST NOTEBOOK
nikola new_post -f ipynb

11 - #EDITAR NOVO POST NOTEBOOK
ipython notebook
#> SALVAR EDIÇÃO no jupyter

12 - #COMPLIAR ALTERAÇÕES
$ nikola build

13 - #PUBLICAR LOCALMENTE
$ nikola serve -b

14 - #DEPLOY EM GITHUB
$ nikola github_deploy

00- VEJA O BLOG EM: https://esloch.github.io/nk-blog/

######################################################################################################
NOTA: >> O branch (gh-pages) é o repositório ao qual o http:// @github.io faz a leitura de index.html
> O branch (gh-pages) é o repositório onde se publica o deploy desde o dev local

> O repositório (master) é onde esta o source com arquivos conf.py, é onde recebe os comandos de new_post

> O comando "$ nikola github_deploy" faz o build, add e commit dos arquivos em git e push para a rama master e publica em (gh-pages)

> O jupyter nao cria desde a interface a tag "date", sendo assim necessária a criação da tag antes de abrir o jupyter. (tem algumas issues a respeito com resolved já...)
